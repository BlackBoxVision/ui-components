# DarkModeToggle

Stories related to **DarkModeToggle** component.

## Example

This is how you can use **DarkModeToggle** component:

```jsx
import React from 'react';
import {
  DarkModeToggle,
  DarkModeProvider,
} from '@blackbox-vision/ui-components';
import { Button, Grid } from '@material-ui/core';

export const MyDarkModeToggle = () => (
  <DarkModeProvider>
    <Grid direction="column" spacing={2} container>
      <Grid item>
        <DarkModeToggle label="Dark mode" />
      </Grid>
      <Grid item>
        <Button color="primary" variant="contained">
          This is button
        </Button>
      </Grid>
    </Grid>
  </DarkModeProvider>
);

MyDarkModeToggle.displayName = 'MyDarkModeToggle';
```
